# Accepting Clients

Let's write a complete server that accepts a single client and reports connection details.

## Complete Accept Example

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <unistd.h>
#include <arpa/inet.h>

#define PORT 8080

int main() {
    // Create socket
    int server_fd = socket(AF_INET, SOCK_STREAM, 0);
    if (server_fd == -1) {
        perror("socket");
        return 1;
    }

    // Set SO_REUSEADDR
    int opt = 1;
    setsockopt(server_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    // Bind
    struct sockaddr_in address;
    memset(&address, 0, sizeof(address));
    address.sin_family = AF_INET;
    address.sin_addr.s_addr = INADDR_ANY;
    address.sin_port = htons(PORT);

    if (bind(server_fd, (struct sockaddr*)&address, sizeof(address)) == -1) {
        perror("bind");
        close(server_fd);
        return 1;
    }

    // Listen
    if (listen(server_fd, 5) == -1) {
        perror("listen");
        close(server_fd);
        return 1;
    }

    printf("Server listening on port %d\n", PORT);
    printf("Waiting for connections...\n\n");

    // Accept a connection
    struct sockaddr_in client_addr;
    socklen_t client_len = sizeof(client_addr);

    int client_fd = accept(server_fd, (struct sockaddr*)&client_addr, &client_len);

    if (client_fd == -1) {
        perror("accept");
        close(server_fd);
        return 1;
    }

    // Report connection details
    char client_ip[INET_ADDRSTRLEN];
    inet_ntop(AF_INET, &client_addr.sin_addr, client_ip, sizeof(client_ip));
    int client_port = ntohs(client_addr.sin_port);

    printf("=== New Connection ===\n");
    printf("Client IP: %s\n", client_ip);
    printf("Client port: %d\n", client_port);
    printf("Client socket fd: %d\n", client_fd);
    printf("======================\n\n");

    // Clean up
    close(client_fd);
    close(server_fd);

    return 0;
}
```

## Getting Socket Information

Use `getpeername()` and `getsockname()` to get address information:

```c
struct sockaddr_in local_addr, peer_addr;

// Get local address (server's address)
socklen_t local_len = sizeof(local_addr);
getsockname(server_fd, (struct sockaddr*)&local_addr, &local_len);

// Get peer address (client's address)
socklen_t peer_len = sizeof(peer_addr);
getpeername(client_fd, (struct sockaddr*)&peer_addr, &peer_len);

printf("Server is at %s:%d\n",
       inet_ntoa(local_addr.sin_addr),
       ntohs(local_addr.sin_port));
printf("Client is at %s:%d\n",
       inet_ntoa(peer_addr.sin_addr),
       ntohs(peer_addr.sin_port));
```

## Multiple Clients

A server typically runs in a loop accepting multiple clients:

```c
while (1) {
    int client_fd = accept(server_fd, (struct sockaddr*)&client_addr, &client_len);

    if (client_fd == -1) {
        perror("accept");
        continue;  // Don't exit, keep accepting
    }

    // Handle this client...
    handle_client(client_fd);

    close(client_fd);
}
```

But this handles one client at a time! We'll explore concurrency in later lessons.
