# Building an Echo Server

Let's put it all together in a complete echo server that handles one client at a time.

## Complete Echo Server

```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <unistd.h>
#include <signal.h>

#define PORT 8080
#define BUFFER_SIZE 1024

volatile sig_atomic_t running = 1;

void handle_sigint(int sig) {
    (void)sig;
    running = 0;
}

void handle_client(int client_fd) {
    char buffer[BUFFER_SIZE];

    printf("Client connected (fd=%d)\n", client_fd);

    while (running) {
        memset(buffer, 0, BUFFER_SIZE);

        ssize_t bytes_read = recv(client_fd, buffer, BUFFER_SIZE - 1, 0);

        if (bytes_read <= 0) {
            if (bytes_read == 0) {
                printf("Client disconnected\n");
            } else {
                perror("recv");
            }
            break;
        }

        printf("Received %zd bytes: %s", bytes_read, buffer);

        // Echo back
        ssize_t bytes_sent = send(client_fd, buffer, bytes_read, 0);

        if (bytes_sent == -1) {
            perror("send");
            break;
        }

        printf("Echoed %zd bytes\n", bytes_sent);
    }

    close(client_fd);
}

int main() {
    // Setup signal handler
    signal(SIGINT, handle_sigint);

    // Create socket
    int server_fd = socket(AF_INET, SOCK_STREAM, 0);
    if (server_fd == -1) {
        perror("socket");
        return 1;
    }

    // Enable address reuse
    int opt = 1;
    setsockopt(server_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    // Bind
    struct sockaddr_in address;
    address.sin_family = AF_INET;
    address.sin_addr.s_addr = INADDR_ANY;
    address.sin_port = htons(PORT);

    if (bind(server_fd, (struct sockaddr*)&address, sizeof(address)) == -1) {
        perror("bind");
        close(server_fd);
        return 1;
    }

    // Listen
    if (listen(server_fd, 5) == -1) {
        perror("listen");
        close(server_fd);
        return 1;
    }

    printf("Echo server listening on port %d\n", PORT);
    printf("Press Ctrl+C to stop\n\n");

    // Accept one client
    struct sockaddr_in client_addr;
    socklen_t client_len = sizeof(client_addr);

    int client_fd = accept(server_fd, (struct sockaddr*)&client_addr, &client_len);

    if (client_fd == -1) {
        perror("accept");
        close(server_fd);
        return 1;
    }

    handle_client(client_fd);

    close(server_fd);
    return 0;
}
```

## Testing

```bash
# Terminal 1: Run server
./echo_server
# Echo server listening on port 8080
# Press Ctrl+C to stop

# Terminal 2: Connect with netcat
nc localhost 8080
# Type: Hello World!
# Server echoes: Hello World!

# Terminal 1 output:
# Client connected (fd=4)
# Received 12 bytes: Hello World!
#
# Echoed 12 bytes
```

## Using telnet

```bash
telnet localhost 8080
# Connected to localhost
# Type anything and press Enter
# Server echoes back
```
