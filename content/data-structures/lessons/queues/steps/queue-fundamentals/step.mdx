# Queue Fundamentals

A queue adds elements at the **rear** and removes from the **front**.

## Array Queue (Linear)

```c
#define MAX_SIZE 100

typedef struct {
    int items[MAX_SIZE];
    int front;
    int rear;
    int size;
} Queue;

void initQueue(Queue *q) {
    q->front = 0;
    q->rear = -1;
    q->size = 0;
}

int isEmpty(Queue *q) {
    return q->size == 0;
}

int isFull(Queue *q) {
    return q->size == MAX_SIZE;
}

void enqueue(Queue *q, int value) {
    if (isFull(q)) {
        printf("Queue overflow\n");
        return;
    }
    q->items[++q->rear] = value;
    q->size++;
}

int dequeue(Queue *q) {
    if (isEmpty(q)) {
        printf("Queue underflow\n");
        return -1;
    }
    q->size--;
    return q->items[q->front++];
}

int peek(Queue *q) {
    if (isEmpty(q)) return -1;
    return q->items[q->front];
}
```

## Circular Queue (Efficient)

```c
typedef struct {
    int items[MAX_SIZE];
    int front;
    int rear;
} CQueue;

void initCQueue(CQueue *q) {
    q->front = -1;
    q->rear = -1;
}

int isCFull(CQueue *q) {
    return (q->rear + 1) % MAX_SIZE == q->front;
}

int isCEmpty(CQueue *q) {
    return q->front == -1;
}

void cenqueue(CQueue *q, int value) {
    if (isCFull(q)) {
        printf("Queue full\n");
        return;
    }
    if (q->front == -1) q->front = 0;
    q->rear = (q->rear + 1) % MAX_SIZE;
    q->items[q->rear] = value;
}

int cdequeue(CQueue *q) {
    if (isCEmpty(q)) {
        printf("Queue empty\n");
        return -1;
    }
    int value = q->items[q->front];
    if (q->front == q->rear) {
        q->front = q->rear = -1;  // Reset queue
    } else {
        q->front = (q->front + 1) % MAX_SIZE;
    }
    return value;
}
```

## Queue as Linked List

```c
typedef struct QNode {
    int data;
    struct QNode *next;
} QNode;

typedef struct {
    QNode *front;
    QNode *rear;
} LinkedQueue;

void enqueueLQ(LinkedQueue *q, int value) {
    QNode *newNode = (QNode *)malloc(sizeof(QNode));
    newNode->data = value;
    newNode->next = NULL;

    if (q->rear == NULL) {
        q->front = q->rear = newNode;
        return;
    }
    q->rear->next = newNode;
    q->rear = newNode;
}

int dequeueLQ(LinkedQueue *q) {
    if (q->front == NULL) return -1;
    QNode *temp = q->front;
    int value = temp->data;
    q->front = q->front->next;
    if (q->front == NULL) q->rear = NULL;
    free(temp);
    return value;
}
```
